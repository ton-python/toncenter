## SDK Usage

```python
import asyncio

from toncenter.rest import ToncenterRestClient
from toncenter.types import Network

LIBRARIES = ["8F452D7A4DFD74066B682365177259ED05734435BE76B5FD4BD5D8AF2B7C3D68"]


async def main() -> None:
    async with ToncenterRestClient(network=Network.MAINNET) as client:
        result = await client.v2.configuration.get_libraries(
            LIBRARIES,
        )
        print(result.model_dump_json(indent=4))


if __name__ == "__main__":
    asyncio.run(main())
```
