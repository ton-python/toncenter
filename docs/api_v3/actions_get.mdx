## SDK Usage

```python
import asyncio

from toncenter.rest import ToncenterRestClient
from toncenter.types import Network

ACCOUNT = "EQDtFpEwcFAEcRe5mLVh2N6C0x-_hJEM7W61_JLnSF74p4q2"


async def main() -> None:
    async with ToncenterRestClient(network=Network.MAINNET) as client:
        result = await client.v3.actions.get_actions(
            account=ACCOUNT,
        )
        print(result.model_dump_json(indent=4))


if __name__ == "__main__":
    asyncio.run(main())
```
