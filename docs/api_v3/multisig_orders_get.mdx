## SDK Usage

```python
import asyncio

from toncenter.rest import ToncenterRestClient
from toncenter.types import Network

MULTISIG_ADDRESS = ["UQCDrgGaI6gWK-qlyw69xWZosurGxrpRgIgSkVsgahUtxZR0"]


async def main() -> None:
    async with ToncenterRestClient(network=Network.MAINNET) as client:
        result = await client.v3.multisig.get_multisig_orders(
            multisig_address=MULTISIG_ADDRESS,
        )
        print(result.model_dump_json(indent=4))


if __name__ == "__main__":
    asyncio.run(main())
```
